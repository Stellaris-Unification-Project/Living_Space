@hab_minerals = 200
@hab_alloys = 150
@hab_maintenance = 2

#########################################################################
district_orbital_habitat_housing = {
	base_buildtime = 240
	icon = district_habitat_housing
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		OR = {
			#is_planet_class = pc_habitat
			is_planet_class = pc_asteroid_habitat
			is_planet_class = pc_ice_asteroid_habitat
			is_planet_class = pc_rare_crystal_asteroid_habitat
			is_planet_class = pc_gas_giant_habitat
		}
	}

	potential = {
		OR = {
			#is_planet_class = pc_habitat
			is_planet_class = pc_asteroid_habitat
			is_planet_class = pc_ice_asteroid_habitat
			is_planet_class = pc_rare_crystal_asteroid_habitat
			is_planet_class = pc_gas_giant_habitat
		}
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = @hab_minerals
			alloys = @hab_alloys
		}
		upkeep = {
			energy = @hab_maintenance
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 8
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 1
		}
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}	
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 9000
			free_housing < 0
		}
		modifier = {
			factor = 0
			free_housing > 1
		}
	}
}
