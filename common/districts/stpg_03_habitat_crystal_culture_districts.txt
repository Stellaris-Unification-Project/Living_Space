	base_buildtime = 240
	is_capped_by_modifier = no

	}

			exists = owner

			}
		}

			job_manager_add = 3
			planet_jobs_unity_produces_mult = 0.03
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner

				is_spiritualist = yes
			}
		}
		modifier = {
			job_priest_add = 3
			planet_jobs_unity_produces_mult = 0.03
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_spiritualist = no
				is_megacorp = no
			}
		}
		text = job_culture_worker_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_spiritualist = no
				is_megacorp = yes
			}
		}
		text = job_manager_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_spiritualist = yes }
		}
		text = job_priest_effect_desc
	}

	ai_weight = {
		weight = 1
		modifier = {
			factor = 3
			free_amenities < 2
		}

		# No one to work it anyway
		modifier = {
			factor = 0
			NOT = {
				any_owned_pop = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant

				}
			}
		}

		trigger = {
			always = yes
		}
	}
